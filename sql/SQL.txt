CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTO_INCREMENT, username VARCHAR(255) UNIQUE, pswd VARCHAR(255), email VARCHAR(255) UNIQUE, date TIMESTAMP );
CREATE TABLE IF NOT EXISTS request_form (
    id INTEGER PRIMARY KEY AUTO_INCREMENT,
    request_type VARCHAR(16) NOT NULL,
    email VARCHAR(255) NOT NULL,
    title VARCHAR(128) NOT NULL,
    description VARCHAR(512) NOT NULL,
    is_picture_included TINYINT(1) NOT NULL,
    area_of_interest VARCHAR(50) NOT NULL
);
